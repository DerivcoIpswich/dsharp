"use strict";

define('test', ['ds'], function(ss) {
  var $global = this;

  // ExpressionTests.Foo

  function Foo(i, j) {
  }
  var Foo$ = {
    toString: function() {
      return 'Foo';
    },
    sum: function(i) {
      return 0;
    }
  };


  // ExpressionTests.Bar

  function Bar(i, j, f) {
    Foo.call(this, i, j);
  }
  var Bar$ = {
    sum: function() {
      return Foo.prototype.sum.call(this, 1) + 1;
    },
    toString: function() {
      return Foo.prototype.toString.call(this) + ' -> Bar';
    }
  };


  var $exports = ds.module('test', null,
    {
      Foo: ds.defineClass(Foo, Foo$, [Number, Number], null),
      Bar: ds.defineClass(Bar, Bar$, [Number, Number, Foo], Foo)
    });


  return $exports;
});
